CREATE DATABASE tkufd DEFAULT CHARACTER SET=utf8mb4 DEFAULT COLLATE=utf8mb4_unicode_ci;

USE tkufd;

CREATE TABLE lang_text (
    lt_id BIGINT UNSIGNED AUTO_INCREMENT NOT NULL,
    lt_label_id BIGINT UNSIGNED NOT NULL,
    lt_lang_id INT UNSIGNED NOT NULL,
    lt_text VARBINARY(255) NOT NULL,
    INDEX lt_label_lang (lt_label_id, lt_lang_id),
    PRIMARY KEY(lt_id)
) ENGINE=InnoDB, DEFAULT CHARSET=binary;

CREATE TABLE bank (
    bank_id BIGINT UNSIGNED AUTO_INCREMENT NOT NULL,
    bank_code VARBINARY(255) NOT NULL,
    bank_label_id BIGINT UNSIGNED NOT NULL,
    INDEX bank_code (bank_code),
    PRIMARY KEY(bank_id)
) ENGINE=InnoDB, DEFAULT CHARSET=binary;

CREATE TABLE bank_data (
    bd_id BIGINT UNSIGNED AUTO_INCREMENT NOT NULL,
    bd_bank_id BIGINT UNSIGNED NOT NULL,
    bd_type INT UNSIGNED NOT NULL,
    bd_timestamp BINARY(14) NOT NULL,
    bd_value BIGINT UNSIGNED NOT NULL,
    INDEX bd_bid_type_timestamp (bd_bank_id, bd_type, bd_timestamp),
    PRIMARY KEY(bd_id)
) ENGINE=InnoDB, DEFAULT CHARSET=binary;
